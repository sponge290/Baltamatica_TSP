## 1. 项目概述

### 1.1 项目背景

本项目旨在构建**全合规、低门槛、高可用**的天气感知型旅行商问题 (TSP) 求解系统。针对传统TSP模型未考虑天气动态影响、公网部署科学计算软件存在授权合规风险的痛点，采用「公网静态前端界面+用户本地北太天元计算」的创新架构，实现考虑时间窗内天气因素动态变化、支持多访问次数约束、允许带环路径解的TSP模型求解，为物流配送、应急救援、无人机路径规划等领域提供符合实际场景、合规可用的决策支持工具。

### 1.2 项目目标

- 实现**适配北太天元v3.6版本**的3种TSP求解算法：动态规划、A*、遗传算法，支持天气因素量化与时间窗口约束

- 构建**Supabase(PostgreSQL)** 数据库，实现基础地理数据、天气数据、测试用例、求解结果的标准化存储与管理

### 4.2 动态规划算法

#### 4.2.1 算法描述

动态规划算法适用于**15个城市以内的小规模**天气感知型TSP问题，通过状态压缩与状态转移求解全局最优路径，适配时间窗口约束与天气动态影响。

#### 4.2.2 北太天元v3.6适配实现

```MATLAB

function [best_path, total_cost, total_time, reliability, nodes, state_process] = DP_TSP(cities, time_windows, weather_data, road_segments)
    % 适配北太天元v3.6 天气感知型TSP动态规划算法
    % 输入：标准化问题参数
    % 输出：最优路径、总成本、总时间、可靠性、节点详情、状态转移过程（可视化用）
    n = size(cities, 1);
    if n > 15
        error('动态规划算法仅支持15个城市以内的小规模问题');
    end
    
    %% 1. 初始化状态表
    INF = 1e18;
    dp = INF * ones(2^n, n);
    dp(1, 1) = 0; % 起点为第1个城市，北太天元索引从1开始
    prev = zeros(2^n, n); % 路径回溯表
    state_process = []; % 状态转移过程，用于D3.js可视化
    
    %% 2. 状态转移
    for mask = 1:2^n-1
        for u = 1:n
            if bitand(mask, bitshift(1, u-1)) ~= 0 && dp(mask, u) < INF
                for v = 1:n
                    if bitand(mask, bitshift(1, v-1)) == 0
                        % 计算当前到达时间
                        current_time = dp(mask, u);
                        % 计算天气感知的旅行时间与成本
                        [travel_time, cost, rel] = calculate_weather_aware_time(u, v, current_time, cities, road_segments, weather_data);
                        % 校验时间窗口约束
                        arrival_time = current_time + travel_time;
                        if is_time_window_valid(v, arrival_time, time_windows)
                            new_mask = bitor(mask, bitshift(1, v-1));
                            if dp(new_mask, v) > dp(mask, u) + cost
                                dp(new_mask, v) = dp(mask, u) + cost;
                                prev(new_mask, v) = u;
                                % 记录状态转移过程，用于可视化
                                state_process = [state_process; mask, new_mask, u, v, cost];
                            end
                        end
                    end
                end
            end
        end
    end
    
    %% 3. 回溯最优路径
    full_mask = 2^n - 1;
    [min_cost, best_idx] = min(dp(full_mask, :) + arrayfun(@(v) calculate_weather_aware_time(v, 1, dp(full_mask, v), cities, road_segments, weather_data), 1:n));
    best_path = reconstruct_path(prev, full_mask, best_idx);
    % 补充返回起点
    best_path = [best_path, 1];
    
    %% 4. 计算路径全量指标与节点详情
    [total_time, reliability, nodes] = calculate_path_metrics(best_path, cities, time_windows, weather_data, road_segments);
    total_cost = min_cost;
end
```

#### 5.1.1 天气数据导入工具

```Python

import pandas as pd
from supabase import create_client, Client
from datetime import datetime

class WeatherDataImporter:
    def __init__(self, supabase_url: str, supabase_key: str):
        # 初始化Supabase客户端
        self.supabase: Client = create_client(supabase_url, supabase_key)

    def import_weather_stations(self, file_path: str):
        """批量导入气象站数据，支持增量更新"""
        df = pd.read_csv(file_path)
        data = df.to_dict('records')
        # 批量写入，冲突时更新
        self.supabase.table("weather_stations").upsert(data, on_conflict="station_id").execute()
        print(f"成功导入{len(data)}条气象站数据")

    def import_weather_observations(self, file_path: str, batch_size: int = 1000):
        """批量导入天气观测数据，分批次写入，避免超时"""
        df = pd.read_csv(file_path)
        # 数据类型标准化
        df['observation_time'] = pd.to_datetime(df['observation_time']).dt.strftime('%Y-%m-%d %H:%M:%S%z')
        # 分批次写入
        for i in range(0, len(df), batch_size):
            batch = df.iloc[i:i+batch_size].to_dict('records')
            self.supabase.table("weather_observations").insert(batch).execute()
            print(f"已导入{i+len(batch)}/{len(df)}条观测数据")

    def close(self):
        """关闭客户端连接"""
        self.supabase.auth.sign_out()
```

2. 前端初始化：用户访问Vercel托管的前端页面，前端从Supabase拉取预设测试用例与基础数据，同时检测本地代理服务连接状态

7. 结果存储与分享：用户点击保存结果，前端将求解结果中转上传至Supabase数据库，生成唯一分享链接，其他用户可通过链接查看可视化结果

#### 7.2.2 核心集成代码示例（前端-本地代理通信）

```JavaScript

import axios from 'axios';
import { createClient } from '@supabase/supabase-js';

// 配置项
const PROXY_BASE_URL = 'http://127.0.0.1:18080';
const supabase = createClient(import.meta.env.VITE_SUPABASE_URL, import.meta.env.VITE_SUPABASE_ANON_KEY);

// 1. 本地代理连接检测
export async function checkProxyConnection() {
  try {
    const res = await axios.get(`${PROXY_BASE_URL}/health`, { timeout: 3000 });
    return { connected: true, version: res.data.version };
  } catch (e) {
    return { connected: false, error: e.message };
  }
}

// 2. 调用北太天元执行TSP算法
export async function runTSPAlgorithm(algorithm, problemData, params) {
  try {
    const res = await axios.post(`${PROXY_BASE_URL}/run`, {
      algorithm,
      problem_data: problemData,
      params
    }, { timeout: 120000 });
    return res.data;
  } catch (e) {
    return { code: 500, msg: '算法执行请求失败', error: e.message };
  }
}

// 3. 求解结果上传Supabase
export async function uploadSolutionResult(caseId, algorithm, resultData) {
  // 插入路径解主表
  const { data: solution, error: solutionError } = await supabase
    .from('route_solutions')
    .insert([{
      case_id: caseId,
      algorithm,
      total_cost: resultData.total_cost,
      total_time: resultData.total_time,
      reliability: resultData.reliability,
      exec_time: resultData.exec_time,
      route_sequence: resultData.path,
      is_public: false
    }])
    .select()
    .single();

  if (solutionError) throw solutionError;

  // 插入路径节点详情
  const { error: nodesError } = await supabase
    .from('route_nodes')
    .insert(resultData.nodes.map((node, index) => ({
      solution_id: solution.solution_id,
      city_id: node.city_id,
      visit_order: index + 1,
      arrival_time: node.arrival_time,
      departure_time: node.departure_time,
      weather_condition: node.weather_condition
    })));

  if (nodesError) throw nodesError;

  return solution.solution_id;
}
```

# 文档二：天气感知型旅行商问题(TSP)模型设计文档（适配版）

核心修改说明：基于原理论模型文档，适配项目实际落地架构，补充模型的工程化实现规范，更新数据库设计以适配Supabase PostgreSQL，重构可视化设计以匹配D3.js前端实现，完善求解算法框架的北太天元v3.6适配规范，让理论模型与实际落地方案完全对齐。

## 1. 引言

旅行商问题 (Traveling Salesman Problem, TSP) 是运筹学和计算机科学中经典的组合优化问题，旨在找到一条最短路径，使得旅行商能够访问所有给定城市并返回起点。传统TSP模型通常假设旅行时间或成本是固定的，不随时间变化。然而，在现实世界中，天气因素对交通系统有着显著影响，降水、大风、低能见度等天气状况会直接导致车辆行驶速度下降、旅行时间增加、行程可靠性降低，使得旅行时间和成本随时间动态变化。

同时，传统TSP求解系统的公网部署，面临着科学计算软件的授权合规、服务器算力成本、高并发性能瓶颈等工程化痛点。本研究在天气感知型TSP理论模型的基础上，提出**「公网静态前端+用户本地计算」**的创新落地方案，构建了一套考虑时间窗内天气因素动态变化、支持多访问次数约束、允许带环路径解的TSP模型，同时解决了模型工程化落地的合规性与成本问题，为物流配送、应急救援、无人机路径规划等领域提供更符合实际情况、可落地使用的决策支持模型与工具。

## 2. 问题描述与假设

### 2.1 问题描述

天气感知型TSP问题可以描述为：给定一组城市（或地点）集合，每个城市有一个或多个可访问的时间窗口，城市间路段的旅行时间与成本随时间和天气状况动态变化，找到一条总成本最小的路径，使得：

1. 每个城市至少被访问指定的最小次数（基础场景为至少1次）

2. 每次访问城市的时间必须落在其对应的有效时间窗口内

3. 路径允许包含环，即可重复访问某些城市，以适配天气变化与时间窗口约束

4. 路段的旅行时间与成本，需基于对应时间的实际天气状况进行动态修正

5. 模型求解过程完全在用户本地环境执行，公网仅提供交互界面与结果可视化能力

### 2.2 基本假设

1. **时间依赖性假设**：城市间的旅行时间和成本随时间与天气状况动态变化，呈现显著的时间依赖性特征。

2. **天气影响可量化假设**：天气因素（降水、风速、能见度等）对路段行驶速度的影响可通过实地观测数据量化为速度折减系数，且该系数具有统计稳定性。

3. **FIFO属性假设**：旅行时间函数满足先进先出（First-In-First-Out, FIFO）属性，即同一路段上，出发时间较晚的车辆不会比出发时间较早的车辆更早到达目的地，避免路径规划出现非因果异常。

4. **时间窗口假设**：每个城市有一个或多个有效时间窗口，仅在时间窗口内访问城市才被视为有效，服务可正常完成。

5. **多次访问假设**：允许多次访问同一城市，以适配不同时间窗口、规避恶劣天气影响、满足多频次服务需求。

6. **本地计算环境假设**：用户本地已安装正版北太天元v3.6及以上版本软件，具备模型求解的本地计算能力，公网环境不提供任何计算服务。

7. **数据可获取假设**：规划区域内的历史/预报天气数据、城市地理数据、路段基础数据可通过公开渠道获取，且可标准化入库。

## 3. 天气因素量化方法

### 3.1 天气数据收集与预处理

#### 3.1.1 数据来源

天气数据可从多个公开合规渠道获取，适配模型的工程化落地：

1. 地面气象站观测数据：从国家气象科学数据中心、NOAA等官方机构获取的逐小时地面气象站观测数据，为模型提供基础天气实况数据。

2. 数值天气预报数据：从ECMWF、GFS等机构获取的72小时逐3小时预报数据，为路径规划提供未来天气变化的预测依据。

3. 交通气象监测数据：从交通部门公开的公路气象监测站获取的路段实况天气数据，提升路段天气影响的量化精度。

#### 3.1.2 数据预处理

收集到的多源天气数据需经过标准化预处理，方可输入模型：

1. **数据清洗**：去除异常值、缺失值、重复数据，对短时间序列缺失值采用线性插值补全，长时间缺失采用空间克里金插值补全。

2. **数据标准化**：将不同单位、不同量级的天气参数转换为标准化指标，统一数据维度。

3. **时空匹配**：通过空间插值方法，将离散气象站的观测数据匹配到具体路段与时间窗口，实现天气数据与路段旅行时间的时空对齐。

4. **数据入库**：预处理完成的数据通过批量导入工具写入Supabase数据库，供前端加载与模型求解调用。

### 3.2 天气因素对旅行时间的影响量化

本模型采用**速度折减系数法**作为核心量化方法，该方法物理意义明确、计算效率高、适配北太天元批量计算，同时可通过统计回归方法进行系数校准，兼顾精度与工程实用性。

#### 3.2.1 核心量化模型

将天气因素的影响量化为对理想天气下自由流速度的折减，核心公式如下：

 $v_{ij}(t, W) = v_{ij}^0 \times f(W(t))$ 

其中：

-  $v_{ij}^0$ ：路段 $ij$ 在理想干燥天气下的自由流速度，由路段限速、道路类型决定

-  $f(W(t))$ ：时间 $t$ 的天气因素综合折减函数，取值范围为 $[0, 1]$ 

-  $v_{ij}(t, W)$ ：考虑天气因素后，路段 $ij$ 在时间 $t$ 的实际行驶速度

基于实际行驶速度，可计算路段的旅行时间：

 $\tau_{ij}(t) = \frac{d_{ij}}{v_{ij}(t, W(t))}$ 

其中：

-  $d_{ij}$ ：路段 $ij$ 的实际里程

-  $\tau_{ij}(t)$ ：时间 $t$ 从城市 $i$ 出发，通过路段 $ij$ 的旅行时间

#### 3.2.2 天气折减系数标定

基于国内公路交通气象观测数据，标定不同天气条件下的速度折减系数，作为模型的基础参数，适配国内道路场景：

|天气条件|速度折减系数|对旅行时间的影响幅度|
|---|---|---|
|干燥晴天|1.0|无影响|
|雾（能见度200-500m）|0.90-0.95|增加5%-10%|
|小雨（日降水量<10mm）|0.85-0.90|增加10%-15%|
|中雨（日降水量10-25mm）|0.80-0.85|增加15%-25%|
|大雨（日降水量>25mm）|0.75-0.85|增加15%-25%|
|小雪（日降雪量<2.5mm）|0.70-0.80|增加20%-30%|
|中雪（日降雪量2.5-5mm）|0.60-0.70|增加30%-40%|
|大雪（日降雪量>5mm）|0.50-0.70|增加30%-50%|
|强逆风（风速>15m/s）|0.80-0.85|增加15%-20%|
|强侧风（风速>15m/s）|0.85-0.90|增加10%-15%|
|强顺风（风速>15m/s）|1.05-1.10|减少5%-10%|
#### 3.2.3 多因素综合折减函数

当存在多种天气因素叠加时，采用乘法模型计算综合折减系数，避免线性叠加导致的过度修正：

 $f(W(t)) = f_{pre}(t) \times f_{wind}(t) \times f_{vis}(t)$ 

其中：

-  $f_{pre}(t)$ ：降水因素折减系数

-  $f_{wind}(t)$ ：风因素折减系数

-  $f_{vis}(t)$ ：能见度因素折减系数

#### 3.2.4 补充量化方法

针对高精度需求场景，模型同时兼容两种补充量化方法：

1. **统计回归模型**：通过线性回归、分位数回归，建立旅行时间与多天气参数的统计关系，适用于有大量历史观测数据的路段。

2. **机器学习模型**：通过随机森林、XGBoost等模型，捕捉天气因素与旅行时间的复杂非线性关系，适用于城市道路等复杂交通场景。

### 3.3 时间依赖型旅行时间函数

为提升模型求解效率，采用**分段常数近似方法**，将规划时间轴划分为多个等长时间段，假设每个时间段内天气状况保持稳定，路段旅行时间为常数：

 $\tau_{ij}(t) = \tau_{ij}^h, \quad t \in [T_h, T_{h+1}]$ 

其中：

-  $\tau_{ij}^h$ ：第 $h$ 个时间段内，路段 $ij$ 的天气修正后旅行时间

-  $[T_h, T_{h+1}]$ ：第 $h$ 个时间段的时间区间，通常取1小时为步长，平衡精度与计算效率

该方法可大幅降低模型求解的计算复杂度，适配北太天元的批量矩阵运算能力，同时满足工程场景的时间精度要求。

## 4. 模型构建

### 4.1 符号定义

|符号|定义|
|---|---|
| $N = {0, 1, ..., n}$ |城市集合，其中0为规划起点/终点|
| $T = [0, T_{max}]$ |规划时间区间|
| $[a_i, b_i]$ |城市 $i$ 的有效访问时间窗口|
| $s_i$ |在城市 $i$ 的服务停留时间|
| $d_{ij}$ |城市 $i$ 到城市 $j$ 的路段里程|
| $\tau_{ij}(t)$ |时间 $t$ 从城市 $i$ 出发，到城市 $j$ 的天气修正后旅行时间|
| $c_{ij}(t)$ |时间 $t$ 从城市 $i$ 出发，到城市 $j$ 的旅行总成本|
| $r_i$ |城市 $i$ 需要被访问的最小次数|
| $x_{ij}(t)$ |0-1决策变量，是否在时间 $t$ 从城市 $i$ 前往城市 $j$ |
| $y_i(t)$ |0-1决策变量，是否在时间 $t$ 访问城市 $i$ |
| $z_i$ |城市 $i$ 的总访问次数|
| $t_i$ |到达城市 $i$ 的时间|
### 4.2 数学模型

#### 目标函数

以总旅行成本最小化为核心优化目标：

 $\min \sum_{i \in N} \sum_{j \in N} \sum_{t \in T} c_{ij}(t) \times x_{ij}(t)$ 

其中，旅行总成本 $c_{ij}(t)$ 由时间成本、燃油成本、可靠性成本共同构成，可根据业务场景调整权重。

#### 约束条件

1. **流量守恒约束**：保证路径的连续性，进入城市的次数等于离开城市的次数

     $\sum_{j \in N} \sum_{t \in T} x_{ij}(t) - \sum_{j \in N} \sum_{t \in T} x_{ji}(t) = 0, \quad \forall i \in N$ 

2. **时间窗口约束**：仅在有效时间窗口内访问城市

     $\sum_{t \in [a_i, b_i]} y_i(t) = z_i, \quad \forall i \in N$ 

3. **访问次数约束**：每个城市的访问次数不低于最小要求

     $z_i \geq r_i, \quad \forall i \in N$ 

4. **旅行时间约束**：保证到达时间的因果关系，满足FIFO属性

     $t_j \geq t_i + s_i + \tau_{ij}(t_i), \quad \forall i,j \in N, x_{ij}(t_i)=1$ 

5. **变量非负与整数约束**

     $x_{ij}(t) \in \{0,1\}, \quad \forall i,j \in N, t \in T \\
y_i(t) \in \{0,1\}, \quad \forall i \in N, t \in T \\
z_i \geq 0, \text{整数}, \quad \forall i \in N$ 

### 4.3 模型扩展

#### 4.3.1 多目标优化扩展

针对多维度决策需求，可将模型扩展为多目标优化问题，同时优化3个核心目标：

 $\min \left[ C, T_{total}, R_{risk} \right]$ 

其中：

-  $C$ ：总旅行成本

-  $T_{total}$ ：总旅行时间

-  $R_{risk}$ ：行程风险指标，由天气恶劣程度、时间窗口违约概率共同决定

采用层次分析法(AHP)、熵权法确定目标权重，将多目标问题转换为单目标问题求解；或采用非支配排序遗传算法(NSGA-II)求解帕累托最优解集，供决策者选择。

#### 4.3.2 不确定性鲁棒扩展

针对天气预报的不确定性，构建鲁棒优化模型，保证路径解在所有可能的天气场景下均满足约束条件：

 $\min \max_{W \in \Omega} \sum_{i \in N} \sum_{j \in N} \sum_{t \in T} c_{ij}(t, W) \times x_{ij}(t)$ 

 $\text{s.t. 所有约束条件在} \forall W \in \Omega \text{ 场景下均成立}$ 

其中， $\Omega$ 为天气因素的不确定集合，基于天气预报的置信区间构建。该模型可有效提升路径规划的抗风险能力，适配应急救援等高可靠性要求的场景。

## 5. 数据库设计（适配Supabase PostgreSQL）

### 5.1 概念模型

本模型的数据库设计围绕天气感知型TSP的核心业务实体展开，分为基础地理数据、天气时序数据、问题配置数据、求解结果数据四大类，核心实体包括：

1. 基础地理实体：城市(City)、路段(RoadSegment)、时间窗口(TimeWindow)

2. 天气数据实体：天气站(WeatherStation)、天气观测(WeatherObservation)、旅行时间(TravelTime)

3. 问题配置实体：预设测试用例(TestCase)

4. 求解结果实体：路径解(RouteSolution)、路径节点(RouteNode)

### 5.2 逻辑模型

### 7.1 动态规划算法框架

适用于**15个城市以内的小规模**天气感知型TSP问题，可求解全局最优解，核心框架适配北太天元v3.6语法：

```MATLAB

function [best_path, total_cost, total_time, reliability, nodes, state_process] = DP_TSP(cities, time_windows, weather_data, road_segments)
    % 北太天元v3.6 天气感知型TSP动态规划算法
    n = size(cities, 1);
    % 状态初始化：北太天元数组索引从1开始
    INF = 1e18;
    dp = INF * ones(2^n, n);
    dp(1, 1) = 0;
    prev = zeros(2^n, n);
    state_process = []; % 可视化用状态转移过程
    
    % 状态转移循环
    for mask = 1:2^n-1
        for u = 1:n
            if bitand(mask, bitshift(1, u-1)) ~= 0 && dp(mask, u) < INF
                for v = 1:n
                    if bitand(mask, bitshift(1, v-1)) == 0
                        % 天气感知旅行时间计算
                        current_time = dp(mask, u);
                        [travel_time, cost, rel] = calculate_weather_aware_time(u, v, current_time, cities, road_segments, weather_data);
                        % 时间窗口约束校验
                        arrival_time = current_time + travel_time;
                        if is_time_window_valid(v, arrival_time, time_windows)
                            new_mask = bitor(mask, bitshift(1, v-1));
                            if dp(new_mask, v) > dp(mask, u) + cost
                                dp(new_mask, v) = dp(mask, u) + cost;
                                prev(new_mask, v) = u;
                                state_process = [state_process; mask, new_mask, u, v, cost];
                            end
                        end
                    end
                end
            end
        end
    end
    
    % 回溯最优路径，补充返回起点
    full_mask = 2^n - 1;
    [min_cost, best_idx] = min(dp(full_mask, :) + arrayfun(@(v) calculate_weather_aware_time(v, 1, dp(full_mask, v), cities, road_segments, weather_data), 1:n));
    best_path = [reconstruct_path(prev, full_mask, best_idx), 1];
    
    % 计算全量指标，用于可视化
    [total_time, reliability, nodes] = calculate_path_metrics(best_path, cities, time_windows, weather_data, road_segments);
    total_cost = min_cost;
end
```
